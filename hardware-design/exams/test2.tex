\begin{itemize}
\item Ask the user how many koalas they have
\item Store their input in a global variable named \texttt{num\_koalas}.
\item Load the value from memory
\item Compute the number of thumbs. Note: each koala has 6 thumbs
\item Report the result
\end{itemize}

\begin{verbatim}
