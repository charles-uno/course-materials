\begin{frame}{For Loop}

\begin{minted}{python}
for x in range(5):
    print(x)
\end{minted}

What does this actually do?

\begin{alltt}
x = 0
loop:
print(x)
x += 1
if x < 5:
    goto loop
\end{alltt}




\end{frame}
